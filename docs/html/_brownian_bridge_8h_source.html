154  struct fn_params {

159  static double fn(double sigma_squared, void *p) {
160  Norm norm;
161  struct fn_params *params = (fn_params *) p;
162  InputIterator it = params->first;
163  NT log_likelihood = 0;
164  while (it != params->beyond) {
165  typename GeometryTraits::MovetkVector v = std::get<ParameterTraits::ParameterColumns::POINT>(*it)
166  - std::get<ParameterTraits::ParameterColumns::MU>(*it);
167  NT operand1 = -LOG_TWO_PI - log(static_cast<NT>(sigma_squared));
168  NT operand2 = -norm(v) / (2 * sigma_squared);
169  log_likelihood += operand1 + operand2;
170  it++;
171  }
172  return -1 * log_likelihood;
173  }
174 
175  void operator()(InputIterator first, InputIterator beyond,
176  NT result, NT x_lower, NT x_upper, NT eps) {
177  struct fn_params params;
178  params.first = first;
179  params.beyond = beyond;
180  F.function = &MLE::fn;
181  F.params = &params;
182  status = gsl_min_fminimizer_set(s, &F, result, x_lower, x_upper);
183  do {
184  iter++;
185  status = gsl_min_fminimizer_iterate(s);
186  result = gsl_min_fminimizer_x_minimum(s);
187  x_lower = gsl_min_fminimizer_x_lower(s);
188  x_upper = gsl_min_fminimizer_x_upper(s);
189 
190  status = gsl_min_test_interval(x_lower, x_upper, eps, 0.0);
191 
192  if (status == GSL_SUCCESS) {
193  estimated_parameter = result;
194  break;
195  }
196  } while (status == GSL_CONTINUE && iter < MaxIter);
197 
198  if (iter >= MaxIter)
199  estimated_parameter = result;
200  }

233  MLE(InputIterator first, InputIterator beyond) {
234  Norm norm;
235  InputIterator it = first;
236  NT upper_bound = 0, squared_length = 0;
237  std::size_t num_elements = std::distance(first, beyond);
238  if (num_elements == 1) {
239  typename GeometryTraits::MovetkVector v = std::get<ParameterTraits::ParameterColumns::POINT>(*first)
240  - std::get<ParameterTraits::ParameterColumns::MU>(*first);
241  NT l = norm(v);
242  if (l < MOVETK_EPS) {
243  estimated_parameter = MOVETK_EPS;
244  return;
245  }
246  }
247  while (it != beyond) {
248  typename GeometryTraits::MovetkVector v = std::get<ParameterTraits::ParameterColumns::POINT>(*it)
249  - std::get<ParameterTraits::ParameterColumns::MU>(*it);
250  NT l = norm(v);
251  squared_length += l;
252  if (l > upper_bound)
253  upper_bound = l;
254  it++;
255  }
256  NT initial_estimate = squared_length / (2 * num_elements);
257  (*this)(first, beyond, initial_estimate, MOVETK_EPS, upper_bound, MOVETK_EPS);
258  }

436  void operator()(InputIterator first, InputIterator beyond, OutputIterator result) {
437  assert(std::distance(first, beyond) >= SIZE);
438  std::vector<typename ParameterTraits::NT> coeffs;
439  std::transform(first, beyond, std::back_inserter(coeffs), [](auto a) {

441  });
442  std::sort(std::begin(coeffs), std::end(coeffs));
443  std::size_t coeff_interval = std::distance(first, beyond) / SIZE;
444  for (std::size_t i = 0; i < SIZE; i++) {
445  std::size_t index = coeff_interval / 2 + coeff_interval * i;
446  *result = coeffs[index];
447  }
448  }
