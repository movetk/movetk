104  template<class TSIterator, class OutputIterator>
105  void operator()(const typename InterpolationTraits::ProbePoint &probe_u,
106  const typename InterpolationTraits::ProbePoint &probe_v,
107  TSIterator first, TSIterator beyond, OutputIterator result) {
108 
109  auto lat_u = std::get<LatIdx>(probe_u);
110  auto lon_u = std::get<LonIdx>(probe_u);
111  typename InterpolationTraits::MovetkPoint p_u = movetk_core::get_point<InterpolationTraits>(lat_u, lon_u,
112  ref);
113 
114  auto lat_v = std::get<LatIdx>(probe_v);
115  auto lon_v = std::get<LonIdx>(probe_v);
116  typename InterpolationTraits::MovetkPoint p_v = movetk_core::get_point<InterpolationTraits>(lat_v, lon_v,
117  ref);
118 
119  auto ts_u = std::get<TsIdx>(probe_u);
120  auto ts_v = std::get<TsIdx>(probe_v);
121 
122  assert(ts_v > ts_u);
123 
124  std::size_t delta_t = ts_v - ts_u;
125  typename InterpolationTraits::NT interval = static_cast<typename InterpolationTraits::NT>(delta_t);
126 
127  std::vector<typename InterpolationTraits::ProbePoint> interpolated_pts;
128 
129  auto tit = first;
130 
131  interpolated_pts.push_back(probe_u);
132 
133  while (++tit != (beyond - 1)) {
134  assert(*tit >= ts_u);
135  assert(*tit <= ts_v);
136  typename InterpolationTraits::ProbePoint probe;
137  delta_t = *tit - ts_u;
138  typename InterpolationTraits::NT fraction = delta_t / interval;
139  typename InterpolationTraits::MovetkVector v = scale(p_u, p_v, fraction);
140  typename InterpolationTraits::MovetkPoint p = translate(p_u, v);
141  typename InterpolationTraits::NT x = movetk_core::get_x<typename InterpolationTraits::GeometryTraits>(
142  p);
143  typename InterpolationTraits::NT y = movetk_core::get_y<typename InterpolationTraits::GeometryTraits>(
144  p);
145  auto geocoordinate = ref.inverse(y, x);
146  std::get<LatIdx>(probe) = geocoordinate[0];
147  std::get<LonIdx>(probe) = geocoordinate[1];
148  std::get<TsIdx>(probe) = *tit;
149  interpolated_pts.push_back(probe);
150  }
151 
152  interpolated_pts.push_back(probe_v);
153 
154  auto ip_first = std::begin(interpolated_pts);
155  auto ip_beyond = std::end(interpolated_pts);
156 
157  if constexpr (SpeedIdx > -1) {
158  std::get<SpeedIdx>(*ip_first) = std::get<SpeedIdx>(probe_u);
159  std::get<SpeedIdx>(*(ip_beyond - 1)) = std::get<SpeedIdx>(probe_v);
160  movetk_core::get_speeds<InterpolationTraits, decltype(ip_first), LatIdx,
161  LonIdx, TsIdx, SpeedIdx>(ip_first, ip_beyond);
162  }
163 
164  if constexpr (HeadingIdx > -1) {
165  std::get<HeadingIdx>(*ip_first) = std::get<HeadingIdx>(probe_u);
166  std::get<HeadingIdx>(*(ip_beyond - 1)) = std::get<HeadingIdx>(probe_v);
167  movetk_core::get_headings<InterpolationTraits, decltype(ip_first), LatIdx,
168  LonIdx, HeadingIdx>(ip_first, ip_beyond);
169  }
170 
171  std::move(ip_first, ip_beyond, result);
172 
173  }
174 
175  };

418  // based on https://doi.org/10.1080/13658816.2014.999682

421  typename InterpolationTraits::Norm norm;
422  typename InterpolationTraits::GeoProjection ref;
423  typename InterpolationTraits::MinimumBoundingRectangle mbr;
424  movetk_core::Scaling<typename InterpolationTraits::GeometryTraits> scale;
425  movetk_core::Translation<typename InterpolationTraits::GeometryTraits> translate;

431  template<class Iterator>
432  Iterator find_source(Iterator first, Iterator current) {
433  Iterator it = current;
434  while (it != first) {
435  typename InterpolationTraits::MovetkVector v = *it - ORIGIN;
436  if ((v * v) > MOVETK_EPS)
437  return it;
438  it--;
439  }
440  return first;
441  }

443  template<class Iterator>
444  Iterator find_destination(Iterator beyond, Iterator current) {
445  Iterator it = current;
446  while (it != beyond) {
447  typename InterpolationTraits::MovetkVector v = *it - ORIGIN;
448  if ((v * v) > MOVETK_EPS)
449  return it;
450  it++;
451  }
452  return (beyond - 1);
453  }

474  template<class TSIterator, class OutputIterator>
475  void operator()(const typename InterpolationTraits::ProbePoint &probe_u,
476  const typename InterpolationTraits::ProbePoint &probe_v,
477  TSIterator first, TSIterator beyond,
478  typename InterpolationTraits::NT max_speed, OutputIterator result) {
479  auto lat_u = std::get<LatIdx>(probe_u);
480  auto lon_u = std::get<LonIdx>(probe_u);
481  typename InterpolationTraits::MovetkPoint p_u = movetk_core::get_point<InterpolationTraits>(lat_u, lon_u,
482  ref);
483 
484  auto lat_v = std::get<LatIdx>(probe_v);
485  auto lon_v = std::get<LonIdx>(probe_v);
486  typename InterpolationTraits::MovetkPoint p_v = movetk_core::get_point<InterpolationTraits>(lat_v, lon_v,
487  ref);
488 
489  auto t_u = std::get<TsIdx>(probe_u);
490  auto t_v = std::get<TsIdx>(probe_v);
491 
492  std::size_t num_elements = std::distance(first, beyond);
493 
494  std::vector<typename InterpolationTraits::ProbePoint> interpolated_pts(num_elements);
495 
496  if (num_elements == 2) {
497  result = probe_u;
498  result = probe_v;
499  return;
500  }
501 
502 
503  std::vector<int> sequence(num_elements - 2);
504  std::iota(std::begin(sequence), std::end(sequence), 1);
505 
506  std::shuffle(std::begin(sequence), std::end(sequence), rng);
507 
508  std::vector<typename InterpolationTraits::MovetkPoint> output(num_elements);
509  std::fill(std::begin(output), std::end(output), ORIGIN);
510 
511  output[0] = p_u;
512  output[num_elements - 1] = p_v;
513  interpolated_pts[0] = probe_u;
514  interpolated_pts[num_elements - 1] = probe_v;
515 
516 
517  bool intersects = false;
518  for (auto &&i : sequence) {
519  auto tit = first + i;
520  auto pit = std::begin(output) + i;
521  auto source = find_source(std::begin(output), pit);
522  auto destination = find_destination(std::end(output), pit);
523  std::size_t position_source = std::distance(std::begin(output), source);
524  std::size_t position_destination = std::distance(std::begin(output), destination);
525  typename InterpolationTraits::NT delta_forward = *tit - *(first + position_source);
526  typename InterpolationTraits::NT delta_backward = *(first + position_destination) - *tit;
527  typename InterpolationTraits::NT radius_u = delta_forward * max_speed;
528  typename InterpolationTraits::NT radius_v = delta_backward * max_speed;
529  typename InterpolationTraits::NT squared_radius_u = radius_u * radius_u;
530  typename InterpolationTraits::NT squared_radius_v = radius_v * radius_v;
531  /*std::cerr<<"source: "<<*source<<"\n";
532  std::cerr<<"destination: "<<*destination<<"\n";
533  std::cerr<<"radius_u: "<<radius_u<<"\n";
534  std::cerr<<"radius_v: "<<radius_v<<"\n";*/
535  typename InterpolationTraits::MovetkVector v = *source - *destination;
536  typename InterpolationTraits::NT squared_length = norm(v);
537  if (squared_length < MOVETK_EPS)
538  continue;
539 
540  auto bounds = mbr(*source, *destination, radius_u, radius_v);
541  typename InterpolationTraits::NT x_min = std::min(
542  movetk_core::get_x<typename InterpolationTraits::GeometryTraits>(bounds.first),
543  movetk_core::get_x<typename InterpolationTraits::GeometryTraits>(bounds.second));
