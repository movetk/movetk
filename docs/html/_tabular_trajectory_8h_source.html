142  template<int timestamp_idx>
143  void insert(ConstTrajectoryIterator pos, TrajectoryIterator first, TrajectoryIterator last) {
144  auto first_ts = std::get<timestamp_idx>(*first);
145  auto last_ts = std::get<timestamp_idx>(*(last - 1));
146 
147  if (!_points.empty()) {
148  if (pos == std::begin(_points)) {
149  auto pos_ts = std::get<timestamp_idx>(*pos);
150  if (last_ts > pos_ts) {
151  throw std::invalid_argument(
152  "Insert (prepend) failed (timestamps of the prepended trajectory should be smaller or equal)!");
153  }
154  } else if (pos == std::end(_points)) {
155  auto before_pos_ts = std::get<timestamp_idx>(*(pos - 1));
156  if (first_ts < before_pos_ts) {
157 // print_tuple(std::cout, *(pos-1));
158 // print_tuple(std::cout, *first);
159  throw std::invalid_argument(
160  "Insert (append) failed (timestamps of the appended trajectory should be greater or equal)!");
161  }
162  } else {
163  auto before_pos_ts = std::get<timestamp_idx>(*(pos - 1));
164  auto pos_ts = std::get<timestamp_idx>(*pos);
165  if (first_ts < before_pos_ts || last_ts > pos_ts) {
166  throw std::invalid_argument(
167  "Insert failed (timestamps of the inserted trajectory are not in the expected range)!");
168  }
169  }
170  }
171 
172  _points.insert(pos, first, last);
173  }

208  TrajectoryIterator operator()(double offset, double &offset_begin, double &offset_end) {
209  auto lower = std::lower_bound(std::begin(offsets), std::end(offsets), offset);
210  if (lower == std::begin(offsets)) {
211  offset_begin = std::numeric_limits<double>::min();
212  offset_end = *lower;
213  } else if (lower == std::end(offsets)) {
214  offset_begin = *(lower - 1);
215  offset_end = std::numeric_limits<double>::max();
216  } else {
217  offset_begin = *(lower - 1);
218  offset_end = *lower;
219  }
220  return _parent->begin() + std::distance(std::begin(offsets), lower);
221  }

227  std::optional<double> reverse_lookup(TrajectoryIterator it) {
228  auto traj_it = _parent->begin();
229  for (auto &offset: offsets) {
230  if (it == traj_it++)

232  }
233  return std::nullopt;

255  template<class GeometryTraits, int LatIdx, int LonIdx, class Interpolator_2>
256  std::pair<value_type, TrajectoryIterator> getInterpolatedPoint(double offset, Interpolator_2 interpolator) {
257  // TODO: assert Interpolator_2 value_type is same as value_type
258  auto offset_fn = lookup_offset_fn<GeometryTraits, LatIdx, LonIdx>();
259  double begin_offset, end_offset;
260  auto seg_end_it = offset_fn(offset, begin_offset, end_offset);
261 
262  if (seg_end_it == std::end(*this)) {
263  return std::make_pair(*(seg_end_it - 1),
264  std::end(*this)); // don't interpolate, return the end point of the trajectory
265  } else if (seg_end_it == std::begin(*this)) {
266  return std::make_pair(*seg_end_it,
267  std::begin(*this)); // don't interpolate, return the start point of the trajectory
268  } else {
269  // interpolator expects the offset between the interpolated points.
270  auto interpolated_point = interpolator(*(seg_end_it - 1), *seg_end_it, offset - begin_offset);
271  return std::make_pair(interpolated_point, seg_end_it);
272  }
273  }

387 template<class... fields, class... new_fields>
388 TabularTrajectory<fields..., new_fields...>
389 concat_field(TabularTrajectory<fields...> t, std::vector<new_fields>... new_columns) {
390  std::array<std::size_t, sizeof...(new_fields)> new_sizes = {new_columns.size()...};
391  std::size_t size = t.size();
392  assert(std::all_of(std::begin(new_sizes), std::end(new_sizes), [&size](std::size_t s) { return s == size; }));
393  std::vector<std::tuple<fields..., new_fields...>> new_data;
394  new_data.reserve(t.size());
395  auto new_it = std::make_tuple(std::begin(new_columns)...);
396  for (auto row_it = std::begin(t); row_it != std::end(t); ++row_it) {
397  auto new_slice = read_row<new_fields...>(new_it);
398  row_next<new_fields...>(new_it);
399  new_data.emplace_back(std::tuple_cat(*row_it, new_slice));
400  }
401  return TabularTrajectory<fields..., new_fields...>(new_data);
402 }
