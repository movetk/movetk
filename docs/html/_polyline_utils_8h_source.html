87  auto lengths = polyline_lengths_m(std::begin(xyzs), std::end(xyzs));
88  std::vector<double> offsets;
89  offsets.reserve(1 + lengths.size());
90  offsets.push_back(0); // add origin offset as well
91  std::partial_sum(std::begin(lengths), std::end(lengths), std::back_inserter(offsets));
92  return offsets;

95  template<class GeometryTraits>
96  struct is_monotone {
97  template<class InputIterator, class OutputIterator>
98  void operator()(InputIterator first, InputIterator beyond,
99  OutputIterator result) {
100  InputIterator cit = first + 2;
101  *result = true;
102  while (cit != beyond) {
103  typename GeometryTraits::MovetkVector u = *(cit - 1) - *(cit - 2);
104  typename GeometryTraits::MovetkVector v = *cit - *(cit - 1);
105  typename GeometryTraits::NT uv = u * v;
106  if (uv < 0) {
107  *result = false;
108  } else {
109  *result = true;
110  }
111  cit++;
112  }
113  }
114  };
