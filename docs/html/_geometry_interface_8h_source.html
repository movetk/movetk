342  template <class OutputIterator,
343  typename = movetk_core::requires_output_iterator<OutputIterator>,
344  typename = movetk_core::requires_tuple<
345  typename OutputIterator::value_type>,
346  typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm>>
347  void operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere,
348  typename IntersectionTraits::GeometryTraits::MovetkSegment &segment, OutputIterator result)
349  {
350  typename IntersectionTraits::Norm norm;
351  typedef typename IntersectionTraits::GeometryTraits::NT NT;
352  typedef typename IntersectionTraits::GeometryTraits::MovetkVector MovetkVector;
353  MovetkVector v1 = segment[1] - segment[0];
354  NT squared_length_v1 = norm(v1);
355  MovetkVector v2 = segment[0] - sphere.center();
356  NT squared_length_v2 = norm(v2);
357  NT q1 = std::pow(v1 * v2, 2) * (1 / squared_length_v1);
358  NT q2 = squared_length_v2 - sphere.squared_radius();
359  if (q1 < q2)
360  result = std::make_tuple(0, -1, -1.0, sphere.center());
361  else if ((q1 - q2) > (MOVETK_EPS * 1000))
362  {
363  NT n = 0.5;
364  NT root = std::pow(squared_length_v1 * (q1 - q2), n);
365  NT translation1 = -1 * (v1 * v2) + root;
366  NT translation2 = -1 * (v1 * v2) - root;
367  MovetkVector v3 = v1;
368  v3 *= (translation1 / squared_length_v1);
369  NT direction = v3 * v1;
370  NT length = norm(v3);
371  NT ratio = length / squared_length_v1;
372  if ((direction > 0) && (ratio <= 1))
373  {
374  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
375  }
376  else if (direction < 0)
377  {
378  result = std::make_tuple(0, -1, ratio, segment[0]);
379  }
380  else if (ratio > 1)
381  {
382  result = std::make_tuple(0, -1, ratio, segment[1]);
383  }
384 
385  v3 *= (squared_length_v1 / translation1);
386  v3 *= (translation2 / squared_length_v1);
387  direction = v3 * v1;
388  length = norm(v3);
389  ratio = length / squared_length_v1;
390  if ((direction > 0) && (ratio <= 1))
391  {
392  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
393  }
394  else if (direction < 0)
395  {
396  result = std::make_tuple(0, -1, ratio, segment[0]);
397  }
398  else if (ratio > 1)
399  {
400  result = std::make_tuple(0, -1, ratio, segment[1]);
401  }
402  }
403  else
404  {
405  NT translation = -1 * (v1 * v2);
406  MovetkVector v3 = v1;
407  v3 *= (translation / squared_length_v1);
408  NT direction = v3 * v1;
409  NT length = norm(v3);
410  NT ratio = length / squared_length_v1;
411  if ((direction > 0) && (ratio <= 1))
412  {
413  result = std::make_tuple(0, 0, ratio, segment[0] + v3);
414  }
415  }
416  }

418  template <typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm>>
419  typename IntersectionTraits::GeometryTraits::MovetkSphere
420  operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_a,
421  typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_b)
422  {
423  // based on https://hal.archives-ouvertes.fr/hal-01955983/document
424  typename IntersectionTraits::Norm norm;
425  typedef typename IntersectionTraits::GeometryTraits GeometryTraits;
426  typedef typename GeometryTraits::NT NT;
427  typedef typename GeometryTraits::MovetkVector MovetkVector;
428  typedef typename GeometryTraits::MovetkPoint MovetkPoint;
429  movetk_core::MakePoint<GeometryTraits> make_point;
430  MakeSphere<GeometryTraits> make_sphere;
431  MovetkPoint ORIGIN = make_point({0, 0});
432  NT squared_r_a = sphere_a.squared_radius();
433  NT r_a = std::pow(squared_r_a, 0.5);
434  NT squared_r_b = sphere_b.squared_radius();
435  NT r_b = std::pow(squared_r_b, 0.5);
436  MovetkVector v = sphere_b.center() - sphere_a.center();
437  MovetkVector v1 = (sphere_b.center() - ORIGIN) +
438  (sphere_a.center() - ORIGIN);
439  NT squared_length = norm(v);
440  NT sum_squared_radius = squared_r_a + squared_r_b;
441  NT squared_diff_radius = sum_squared_radius - 2 * r_a * r_b;
442  NT squared_sum_radius = sum_squared_radius + 2 * r_a * r_b;
443  bool disjoint = (squared_sum_radius - squared_length < 0) || (squared_diff_radius - squared_length > 0);
444  if (!disjoint)
445  {
446  v *= (squared_r_a - squared_r_b) / (2 * squared_length);
447  v1 *= 0.5;
448  MovetkVector v2 = v1 + v;
449  MovetkPoint center = ORIGIN + v2;
450  NT op1 = (squared_r_a + squared_r_b) * 0.5;
451  NT numerator = (squared_r_a - squared_r_b) * (squared_r_a - squared_r_b);
452  NT op2 = 0.25 * numerator / squared_length;
453  NT squared_radius = op1 - op2 - 0.25 * squared_length;
454  return make_sphere(center, squared_radius, false);
455  }
456  else
457  {
458  return make_sphere(ORIGIN, 0);
459  }
460  }

468  template <class GeometryTraits,
469  class Norm,
470  typename = movetk_core::requires_planar_geometry<GeometryTraits>,
471  typename = movetk_core::requires_L2_norm<Norm>>
472  class Wedge
473  {
474  private:
475  typedef typename GeometryTraits::NT NT;
476  Norm norm;
477  movetk_core::MakePoint<GeometryTraits> make_point;
478  typename GeometryTraits::MovetkPoint ORIGIN = make_point({0, 0});
479  typename GeometryTraits::MovetkVector e1 = make_point({1, 0}) - ORIGIN;
480  typename GeometryTraits::MovetkVector e2 = make_point({0, 1}) - ORIGIN;
481  typename GeometryTraits::MovetkVector _slope = ORIGIN - ORIGIN;
482  typename GeometryTraits::MovetkVector _intercept = ORIGIN - ORIGIN;
483  bool horizontal = false, vertical = false;
484  bool degenerate = false;
485  bool upper_right = false, lower_left = false;
486  bool lower_right = false, upper_left = false;
487 
488  public:
489  Wedge() = default;
490 
491  Wedge(typename GeometryTraits::MovetkPoint &p,
492  typename GeometryTraits::MovetkPoint &center,
493  typename GeometryTraits::NT radius)
494  {
495  NT m1, m2, c1, c2, tanA, tanB;
496  NT squared_radius = radius * radius;
497  typename GeometryTraits::MovetkVector _slope_ray = center - p;
498  NT v_x = _slope_ray * e1;
499  NT v_y = _slope_ray * e2;
500  NT segment_squared_length = norm(_slope_ray);
501  NT root = norm ^ 1;
502  if (segment_squared_length < squared_radius)
503  {
504  degenerate = true;
505  }
506  if (root < MOVETK_EPS)
507  {
508  degenerate = true;
509  }
510  if (abs(v_x) < MOVETK_EPS)
511  {
512  vertical = true;
513  }
514  if (abs(v_y) < MOVETK_EPS)
515  {
516  horizontal = true;
517  }
518  if (v_x >= 0 && v_y >= 0)
519  {
520  upper_right = true;
521  }
522  if (v_y <= 0 && v_x >= 0)
523  {
524  lower_right = true;
525  }
526  if (v_y >= 0 && v_x <= 0)
527  {
528  upper_left = true;
529  }
530  if (v_x <= 0 && v_y <= 0)
531  {
532  lower_left = true;
533  }
534  NT tangent_squared_length = segment_squared_length - squared_radius;
535  if (abs(tangent_squared_length) < MOVETK_EPS)
536  degenerate = true;
537  else
538  tanB = radius / sqrt(tangent_squared_length);
539  if (degenerate)
540  {
541  _slope = ORIGIN - ORIGIN;
542  }
543  else if (horizontal)
544  {
545  if (upper_right || lower_right)
546  _slope = make_point({tanB, -tanB}) - ORIGIN;
547  else if (upper_left || lower_left)
548  _slope = make_point({-tanB, tanB}) - ORIGIN;
549  }
550  else if (vertical)
551  {
552  tanA = tan(PI / 2 - MOVETK_EPS * 0.001);
553  if ((tanA * tanB) == 1)
554  m1 = (tanA + tanB) / MOVETK_EPS;
555  else
556  m1 = (tanA + tanB) / (1 - tanA * tanB);
557  m2 = -m1;
558  _slope = make_point({m1, m2}) - ORIGIN;
559  }
560  else
561  {
562  tanA = v_y / v_x; // slope of the ray
563  if ((tanA * tanB) == 1)
564  {
565  m1 = (tanA + tanB) / MOVETK_EPS;
566  m2 = (tanA - tanB) / (1 + tanA * tanB);
567  }
568  else if ((tanA * tanB) == -1)
569  {
570  m1 = (tanA + tanB) / (1 - tanA * tanB);
571  m2 = (tanA - tanB) / (-1 * MOVETK_EPS);
572  }
573  else
574  {
575  m1 = (tanA + tanB) / (1 - tanA * tanB);
576  m2 = (tanA - tanB) / (1 + tanA * tanB);
577  }
578 
579  if (upper_right || lower_right)
580  _slope = make_point({m1, m2}) - ORIGIN;
581  else if (upper_left || lower_left)
582  _slope = make_point({m2, m1}) - ORIGIN;
583  }
584 
585  c1 = (make_point({-1 * (_slope * e1), 1}) - ORIGIN) * (p - ORIGIN); // (y - y0) = m * (x - x0)
586  c2 = (make_point({-1 * (_slope * e2), 1}) - ORIGIN) * (p - ORIGIN);
587  _intercept = make_point({c1, c2}) - ORIGIN;
588  }
589 
590  Wedge(typename GeometryTraits::MovetkVector &slope,
591  typename GeometryTraits::MovetkVector &intercept) : _slope(slope),
592  _intercept(intercept) {}
593 
594  typename GeometryTraits::MovetkVector &slope()
595  {
596  return _slope;
597  }
598 
599  typename GeometryTraits::MovetkVector &intercept()
600  {
601  return _intercept;
602  }
603 
604  bool is_empty()
605  {
606  if (_slope == (ORIGIN - ORIGIN))
607  {
608  if (_intercept == (ORIGIN - ORIGIN))
609  {
610  return true;
611  }
612  }
613  return false;
614  }

621  Wedge operator*(Wedge &w)
622  {
623  constexpr std::size_t size = 2 * Norm::P;
624  std::array<std::size_t, size> positions = {0, 1, 2, 3};
625  std::size_t sum = *(positions.end() - 1) + *(positions.end() - 2);
626  typename GeometryTraits::MovetkVector that_slope = w.slope();
627  typename GeometryTraits::MovetkVector that_intercept = w.intercept();
628  std::array<NT, size> slopes = {this->_slope * e1, this->_slope * e2,
629  that_slope * e1, that_slope * e2};
630  std::array<NT, size> intercepts = {this->_intercept * e1, this->_intercept * e2,
631  that_intercept * e1, that_intercept * e2};
632 
633  std::sort(std::begin(positions), std::end(positions), [&slopes](size_t i, size_t j) {
634  return slopes[i] < slopes[j];
635  });
636 
637  auto start = positions.cbegin();
638  auto end = positions.cend();
639  auto sit = slopes.cbegin();
640  auto it = intercepts.cbegin();
641 
642  NT diff_this = (slopes[0] - slopes[1]) / (1 + (slopes[0] * slopes[1]));
643  NT diff_that = (slopes[2] - slopes[3]) / (1 + (slopes[2] * slopes[3]));
644  NT WedgeAngle_this = 2 * movetk_core::rad2deg(diff_this);
645  NT WedgeAngle_that = 2 * movetk_core::rad2deg(diff_that);
646 
647  if ((*start + *(start + 1) == sum) || (*(end - 1) + *(end - 2) == sum))
648  {
649  if ((abs(WedgeAngle_this) < 90) || (abs(WedgeAngle_that) < 90))
650  return Wedge();
651  }
652 
653  if (slopes[0] > slopes[1])
654  {
655  that_slope = make_point({*(sit + *(start + 2)), *(sit + *(start + 1))}) - ORIGIN;
656  that_intercept = make_point({*(it + *(start + 2)), *(it + *(start + 1))}) - ORIGIN;
657  }
658  else
659  {
660  that_slope = make_point({*(sit + *(start + 1)), *(sit + *(start + 2))}) - ORIGIN;
661  that_intercept = make_point({*(it + *(start + 1)), *(it + *(start + 2))}) - ORIGIN;
662  }
663  return Wedge(that_slope, that_intercept);
664  }

671  bool operator*(typename GeometryTraits::MovetkPoint &p)
672  {
673  NT mx1 = (make_point({this->_slope * this->e1, -1}) - ORIGIN) * (p - ORIGIN);
674  NT mx2 = (make_point({this->_slope * this->e2, -1}) - ORIGIN) * (p - ORIGIN);
675  typename GeometryTraits::MovetkVector v = make_point({mx1, mx2}) - ORIGIN;
676  typename GeometryTraits::MovetkVector result = v + this->_intercept;
677 
678  if ((result * this->e1) >= 0 && (result * this->e2) <= 0)
679  return true;
680 
681  if ((result * this->e1) <= 0 && (result * this->e2) >= 0)
682  return true;
683 
684  return false;
685  }
