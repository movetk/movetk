344  template<class OutputIterator,
345  typename = movetk_core::requires_output_iterator<OutputIterator>,
346  typename = movetk_core::requires_tuple<
347  typename OutputIterator::value_type>,
348  typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm > >
349  void operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere,
350  typename IntersectionTraits::GeometryTraits::MovetkSegment &segment, OutputIterator result) {
351  typename IntersectionTraits::Norm norm;
352  typedef typename IntersectionTraits::GeometryTraits::NT NT;
353  typedef typename IntersectionTraits::GeometryTraits::MovetkVector MovetkVector;
354  MovetkVector v1 = segment[1] - segment[0];
355  NT squared_length_v1 = norm(v1);
356  MovetkVector v2 = segment[0] - sphere.center();
357  NT squared_length_v2 = norm(v2);
358  NT q1 = std::pow(v1 * v2, 2) * (1 / squared_length_v1);
359  NT q2 = squared_length_v2 - sphere.squared_radius();
360  if (q1 < q2)
361  result = std::make_tuple(0, -1, -1.0, sphere.center());
362  else if ((q1 - q2) > (MOVETK_EPS * 1000)) {
363  NT n = 0.5;
364  NT root = std::pow(squared_length_v1 * (q1 - q2), n);
365  NT translation1 = -1 * (v1 * v2) + root;
366  NT translation2 = -1 * (v1 * v2) - root;
367  MovetkVector v3 = v1;
368  v3 *= (translation1 / squared_length_v1);
369  NT direction = v3 * v1;
370  NT length = norm(v3);
371  NT ratio = length / squared_length_v1;
372  if ((direction > 0) && (ratio <= 1)) {
373  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
374  } else if (direction < 0) {
375  result = std::make_tuple(0, -1, ratio, segment[0]);
376  } else if (ratio > 1) {
377  result = std::make_tuple(0, -1, ratio, segment[1]);
378  }
379 
380  v3 *= (squared_length_v1 / translation1);
381  v3 *= (translation2 / squared_length_v1);
382  direction = v3 * v1;
383  length = norm(v3);
384  ratio = length / squared_length_v1;
385  if ((direction > 0) && (ratio <= 1)) {
386  result = std::make_tuple(0, 1, ratio, segment[0] + v3);
387  } else if (direction < 0) {
388  result = std::make_tuple(0, -1, ratio, segment[0]);
389  } else if (ratio > 1) {
390  result = std::make_tuple(0, -1, ratio, segment[1]);
391  }
392 
393  } else {
394  NT translation = -1 * (v1 * v2);
395  MovetkVector v3 = v1;
396  v3 *= (translation / squared_length_v1);
397  NT direction = v3 * v1;
398  NT length = norm(v3);
399  NT ratio = length / squared_length_v1;
400  if ((direction > 0) && (ratio <= 1)) {
401  result = std::make_tuple(0, 0, ratio, segment[0] + v3);
402  }
403  }
404  }

406  template<typename = movetk_core::requires_L2_norm<typename IntersectionTraits::Norm> >
407  typename IntersectionTraits::GeometryTraits::MovetkSphere
408  operator()(typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_a,
409  typename IntersectionTraits::GeometryTraits::MovetkSphere &sphere_b) {
410  // based on https://hal.archives-ouvertes.fr/hal-01955983/document
411  typename IntersectionTraits::Norm norm;
412  typedef typename IntersectionTraits::GeometryTraits GeometryTraits;
413  typedef typename GeometryTraits::NT NT;
414  typedef typename GeometryTraits::MovetkVector MovetkVector;
415  typedef typename GeometryTraits::MovetkPoint MovetkPoint;
416  movetk_core::MakePoint<GeometryTraits> make_point;
417  MakeSphere<GeometryTraits> make_sphere;
418  MovetkPoint ORIGIN = make_point({0, 0});
419  NT squared_r_a = sphere_a.squared_radius();
420  NT r_a = std::pow(squared_r_a, 0.5) ;
421  NT squared_r_b = sphere_b.squared_radius();
422  NT r_b = std::pow(squared_r_b, 0.5) ;
423  MovetkVector v = sphere_b.center() - sphere_a.center();
424  MovetkVector v1 = (sphere_b.center() - ORIGIN) +
425  (sphere_a.center() - ORIGIN);
426  NT squared_length = norm(v);
427  NT sum_squared_radius = squared_r_a + squared_r_b;
428  NT squared_diff_radius = sum_squared_radius - 2 * r_a * r_b;
429  NT squared_sum_radius = sum_squared_radius + 2 * r_a * r_b;
430  bool disjoint = (squared_sum_radius - squared_length < 0) || (squared_diff_radius - squared_length > 0);
431  if (!disjoint) {
432  v *= (squared_r_a - squared_r_b) / (2 * squared_length);
433  v1 *= 0.5;
434  MovetkVector v2 = v1 + v;
435  MovetkPoint center = ORIGIN + v2;
436  NT op1 = (squared_r_a + squared_r_b) * 0.5;
437  NT numerator = (squared_r_a - squared_r_b) * (squared_r_a - squared_r_b);
438  NT op2 = 0.25 * numerator / squared_length;
439  NT squared_radius = op1 - op2 - 0.25 * squared_length;
440  return make_sphere(center, squared_radius, false);
441  } else {
442  return make_sphere(ORIGIN, 0);
443  }
444 
445  }

499  template<class DistanceMatrix, class InputIterator,
500  typename = movetk_core::requires_random_access_iterator<InputIterator>,
501  typename = movetk_core::requires_random_access_iterator<typename InputIterator::value_type::iterator>,
502  typename = movetk_core::requires_movetk_point<GeometryTraits, typename InputIterator::value_type::value_type>,
503  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::iterator>,
504  typename = movetk_core::requires_random_access_iterator<typename DistanceMatrix::value_type::iterator>,
505  typename = movetk_core::requires_NT<GeometryTraits, typename DistanceMatrix::value_type::value_type> >
506  typename GeometryTraits::NT
507  operator()(InputIterator first, InputIterator beyond, DistanceMatrix &upper_triag_mat) {
508  typedef discrete_frechet_distance_algorithm<GeometryTraits, Norm,
509  typename GeometryTraits::MovetkDiscreteFrechetDistance> algorithm;
510  typename algorithm::discrete_frechet_distance distance;
511  movetk_core::movetk_back_insert_iterator bit(upper_triag_mat);
512  auto it = first;
513  while (it != beyond) {
514  auto nit = it + 1;
515  typename DistanceMatrix::value_type node;
516  movetk_core::movetk_back_insert_iterator node_bit(node);
517  while (nit != beyond) {
518  node_bit = distance(std::cbegin(*it), std::cend(*it), std::cbegin(*nit), std::cend(*nit));
519  nit++;
520  }
521  bit = node;
522  it++;
523  }
524  auto rit = std::begin(upper_triag_mat);
525  typename GeometryTraits::NT min_sum = 1 / MOVETK_EPS;
526  std::size_t num_columns = std::distance(std::begin(*rit), std::end(*rit));
527  std::size_t row_id = 0;
528  while (rit != std::end(upper_triag_mat)) {
529  typename GeometryTraits::NT max = MOVETK_EPS;
530  typename GeometryTraits::NT second_max = MOVETK_EPS;
531  auto vit = std::begin(*rit);
532  std::size_t num_cols_right = std::distance(std::begin(*rit), std::end(*rit));
533  std::size_t num_cols_left = num_columns - num_cols_right;
534  std::size_t counter = 0;
535  while (counter != num_cols_left) {
536  auto row_it = std::begin(upper_triag_mat) + counter;
537  std::size_t offset = row_id - counter - 1;
538  auto column_it = std::begin(*row_it) + offset;
539  if (*column_it > max) {
540  second_max = max;
541  max = *column_it;
542  } else if (*column_it > second_max && *column_it < max) {
543  second_max = *column_it;
544  }
545  counter++;
546  }
547  while (vit != std::end(*rit)) {
548  if (*vit > max) {
549  second_max = max;
550  max = *vit;
551  } else if (*vit > second_max && *vit < max) {
552  second_max = *vit;
553  }
554  vit++;
555  }
556  typename GeometryTraits::NT sum = max + second_max;
557  if (sum < min_sum) {
558  min_sum = sum;
559  }
560  rit++;
561  row_id++;
562  }
563  return min_sum;
564  }
565  };

573  template<class GeometryTraits,
574  class Norm,
575  typename = movetk_core::requires_planar_geometry<GeometryTraits>,
576  typename = movetk_core::requires_L2_norm<Norm> >
577  class Wedge {
578  private:
579  typedef typename GeometryTraits::NT NT;
580  Norm norm;
581  movetk_core::MakePoint<GeometryTraits> make_point;
582  typename GeometryTraits::MovetkPoint ORIGIN = make_point({0, 0});
583  typename GeometryTraits::MovetkVector e1 = make_point({1, 0}) - ORIGIN;
584  typename GeometryTraits::MovetkVector e2 = make_point({0, 1}) - ORIGIN;
585  typename GeometryTraits::MovetkVector _slope = ORIGIN - ORIGIN;
586  typename GeometryTraits::MovetkVector _intercept = ORIGIN - ORIGIN;
587  bool horizontal = false, vertical = false;
588  bool degenerate = false;
589  bool upper_right = false, lower_left = false;
590  bool lower_right = false, upper_left = false;
591  public:
592  Wedge() = default;
593 
594  Wedge(typename GeometryTraits::MovetkPoint &p,
595  typename GeometryTraits::MovetkPoint &center,
596  typename GeometryTraits::NT radius) {
597  NT m1, m2, c1, c2, tanA, tanB;
598  NT squared_radius = radius * radius;
599  typename GeometryTraits::MovetkVector _slope_ray = center - p;
600  NT v_x = _slope_ray * e1;
601  NT v_y = _slope_ray * e2;
602  NT segment_squared_length = norm(_slope_ray);
603  NT root = norm ^1;
604  if (segment_squared_length < squared_radius) {
605  degenerate = true;
606  }
607  if (root < MOVETK_EPS) {
608  degenerate = true;
609  }
610  if (abs(v_x) < MOVETK_EPS) {
611  vertical = true;
612  }
613  if (abs(v_y) < MOVETK_EPS) {
614  horizontal = true;
615  }
616  if (v_x >= 0 && v_y >= 0) {
617  upper_right = true;
618  }
619  if (v_y <= 0 && v_x >= 0) {
620  lower_right = true;
621  }
622  if (v_y >= 0 && v_x <= 0) {
623  upper_left = true;
624  }
625  if (v_x <= 0 && v_y <= 0) {
626  lower_left = true;
627  }
628  NT tangent_squared_length = segment_squared_length - squared_radius;
629  if (abs(tangent_squared_length) < MOVETK_EPS)
630  degenerate = true;
631  else
632  tanB = radius / sqrt(tangent_squared_length);
633  if (degenerate) {
634  _slope = ORIGIN - ORIGIN;
635  } else if (horizontal) {
636  if (upper_right || lower_right)
637  _slope = make_point({tanB, -tanB}) - ORIGIN;
638  else if (upper_left || lower_left)
639  _slope = make_point({-tanB, tanB}) - ORIGIN;
640  } else if (vertical) {
641  tanA = tan(PI / 2 - MOVETK_EPS * 0.001);
642  if ((tanA * tanB) == 1)
643  m1 = (tanA + tanB) / MOVETK_EPS;
644  else
645  m1 = (tanA + tanB) / (1 - tanA * tanB);
646  m2 = -m1;
647  _slope = make_point({m1, m2}) - ORIGIN;
648  } else {
649  tanA = v_y / v_x; // slope of the ray
650  if ((tanA * tanB) == 1) {
651  m1 = (tanA + tanB) / MOVETK_EPS;
652  m2 = (tanA - tanB) / (1 + tanA * tanB);
653  } else if ((tanA * tanB) == -1) {
654  m1 = (tanA + tanB) / (1 - tanA * tanB);
655  m2 = (tanA - tanB) / (-1 * MOVETK_EPS);
656  } else {
657  m1 = (tanA + tanB) / (1 - tanA * tanB);
658  m2 = (tanA - tanB) / (1 + tanA * tanB);
659  }
660 
661  if (upper_right || lower_right)
662  _slope = make_point({m1, m2}) - ORIGIN;
663  else if (upper_left || lower_left)
664  _slope = make_point({m2, m1}) - ORIGIN;
665 
666  }
667 
668  c1 = (make_point({-1 * (_slope * e1), 1}) - ORIGIN) * (p - ORIGIN); // (y - y0) = m * (x - x0)

700  Wedge operator*(Wedge &w) {
701  constexpr std::size_t size = 2 * Norm::P;
702  std::array<std::size_t, size> positions = {0, 1, 2, 3};
703  std::size_t sum = *(positions.end() - 1) + *(positions.end() - 2);
704  typename GeometryTraits::MovetkVector that_slope = w.slope();
705  typename GeometryTraits::MovetkVector that_intercept = w.intercept();
706  std::array<NT, size> slopes = {this->_slope * e1, this->_slope * e2,
707  that_slope * e1, that_slope * e2};
708  std::array<NT, size> intercepts = {this->_intercept * e1, this->_intercept * e2,
709  that_intercept * e1, that_intercept * e2};
710 
711  std::sort(std::begin(positions), std::end(positions), [&slopes](size_t i, size_t j) {
712  return slopes[i] < slopes[j];
713  });
714 
715  auto start = positions.cbegin();
716  auto end = positions.cend();
717  auto sit = slopes.cbegin();
718  auto it = intercepts.cbegin();
719 
720  NT diff_this = (slopes[0] - slopes[1]) / (1 + (slopes[0] * slopes[1]));
721  NT diff_that = (slopes[2] - slopes[3]) / (1 + (slopes[2] * slopes[3]));
722  NT WedgeAngle_this = 2 * movetk_core::rad2deg(diff_this);
723  NT WedgeAngle_that = 2 * movetk_core::rad2deg(diff_that);
724 
725  if ((*start + *(start + 1) == sum) || (*(end - 1) + *(end - 2) == sum)) {
726  if ((abs(WedgeAngle_this) < 90) || (abs(WedgeAngle_that) < 90))
727  return Wedge();
728  }
729 
730  if (slopes[0] > slopes[1]) {
731  that_slope = make_point({*(sit + *(start + 2)), *(sit + *(start + 1))}) - ORIGIN;
732  that_intercept = make_point({*(it + *(start + 2)), *(it + *(start + 1))}) - ORIGIN;
733  } else {
734  that_slope = make_point({*(sit + *(start + 1)), *(sit + *(start + 2))}) - ORIGIN;
735  that_intercept = make_point({*(it + *(start + 1)), *(it + *(start + 2))}) - ORIGIN;
736  }
737  return Wedge(that_slope, that_intercept);
738  }

745  bool operator*(typename GeometryTraits::MovetkPoint &p) {
746  NT mx1 = (make_point({this->_slope * this->e1, -1}) - ORIGIN) * (p - ORIGIN);
747  NT mx2 = (make_point({this->_slope * this->e2, -1}) - ORIGIN) * (p - ORIGIN);
748  typename GeometryTraits::MovetkVector v = make_point({mx1, mx2}) - ORIGIN;
749  typename GeometryTraits::MovetkVector result = v + this->_intercept;

752  return true;
753 
754  if ((result * this->e1) <= 0 && (result * this->e2) >= 0)
755  return true;
756 
757  return false;
758  }
